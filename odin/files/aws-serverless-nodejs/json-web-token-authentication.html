JSON Web Token

To add authentication to your Express API, the most common and production-ready approach is JWT (JSON Web Token) authentication.

I'll show you a clean setup that works well with your AWS serverless backend.

✅ 1. Install dependencies


npm install jsonwebtoken

✅ 2. Create Auth Middleware

Create a file: middleware/auth.js


const jwt = require('jsonwebtoken');

module.exports = function (req, res, next) {
  try {
    const token = req.headers['authorization'];

    if (!token) {
      return res.status(401).json({
        status: false,
        message: 'No token provided'
      });
    }

    // Remove "Bearer "
    const bearerToken = token.split(' ')[1];

    const decoded = jwt.verify(bearerToken, process.env.JWT_SECRET);

    req.user = decoded; // attach user data
    next();
  } catch (error) {
    return res.status(401).json({
      status: false,
      message: 'Invalid or expired token'
    });
  }
};

✅ 3. Add a Login Route (Generate Token)

Example: routes/auth.js


const express = require('express');
const jwt = require('jsonwebtoken');

const router = express.Router();

// Dummy login (replace with DB check)
router.post('/login', async (req, res) => {
  const { email } = req.body;

  // TODO: validate user from DB
  const user = { id: 1, email };

  const token = jwt.sign(user, process.env.JWT_SECRET, {
    expiresIn: '7d'
  });

  res.json({
    status: true,
    token
  });
});

module.exports = router;

✅ 4. Protect Your Routes

Update your main app:


const express = require('express');
const app = express();

const authMiddleware = require('./middleware/auth');
const authRoutes = require('./routes/auth');

app.use(express.json());

// Public route
app.get('/', (req, res) => {
  res.json({ message: 'aethercorp api is running 🚀' });
});

// Auth routes
app.use('/auth', authRoutes);

// Protected route
app.get('/secure', authMiddleware, (req, res) => {
  res.json({
    message: 'Secure data 🔐',
    user: req.user
  });
});

module.exports = app;

✅ 5. Add Environment Variable


JWT_SECRET=supersecretkey

(For AWS Lambda → use environment variables in Serverless Framework)

You should add JWT_SECRET under the provider.environment section in your serverless.yml.

Here’s the correct way to do it 👇

updated serverless.yml


service: serverless-express-api

provider:
    name: aws
    runtime: nodejs24.x
    region: ap-southeast-1
    memorySize: 128
    timeout: 10

    environment:
    JWT_SECRET: ${env:JWT_SECRET}

functions:
    api:
    handler: handler.handler
    events:
        - httpApi:
            path: /{proxy+}
            method: ANY

plugins:
    - serverless-offline

✅ 6. How It Works (Flow)

  1. Client logs in → /auth/login
  2. Server returns JWT token
  3. Client sends token in header: 
    
Authorization: Bearer <your_token_here>
  4. Middleware verifies token
  5. If valid → access granted
