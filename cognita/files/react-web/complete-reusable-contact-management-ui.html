‚úÖ ContactCard.jsx

A reusable display card for a contact.


<script type="text/jsx">
import React from "react";
import { Card, CardContent, Typography, IconButton, Stack, Avatar } from "@mui/material";
import { Edit, Delete } from "@mui/icons-material";

const ContactCard = ({ contact, onEdit, onDelete }) => {
  return (
    <Card variant="outlined" sx={{ borderRadius: 3, p: 1, boxShadow: 2 }}>
      <CardContent>
        <Stack direction="row" alignItems="center" spacing={2}>
          <Avatar sx={{ bgcolor: "primary.main" }}>
            {contact.name?.charAt(0).toUpperCase()}
          </Avatar>
          <Stack flex={1}>
            <Typography variant="subtitle1" fontWeight={600}>
              {contact.name}
            </Typography>
            <Typography variant="body2" color="text.secondary">
              {contact.email}
            </Typography>
            <Typography variant="body2" color="text.secondary">
              {contact.phone}
            </Typography>
          </Stack>
          <Stack direction="row" spacing={1}>
            <IconButton color="primary" onClick={() => onEdit(contact)}>
              <Edit />
            </IconButton>
            <IconButton color="error" onClick={() => onDelete(contact)}>
              <Delete />
            </IconButton>
          </Stack>
        </Stack>
      </CardContent>
    </Card>
  );
};

export default ContactCard;
</script>

‚úÖ ContactForm.jsx

Reusable form component for adding or editing a contact.


<script type="text/jsx">
import React, { useState, useEffect } from "react";
import { Dialog, DialogTitle, DialogContent, DialogActions, TextField, Button, Stack } from "@mui/material";

const ContactForm = ({ open, onClose, onSave, contact }) => {
  const [form, setForm] = useState({ name: "", email: "", phone: "" });

  useEffect(() => {
    if (contact) setForm(contact);
    else setForm({ name: "", email: "", phone: "" });
  }, [contact]);

  const handleChange = (e) => {
    setForm({ ...form, [e.target.name]: e.target.value });
  };

  const handleSubmit = () => {
    onSave(form);
    onClose();
  };

  return (
    <Dialog open={open} onClose={onClose} fullWidth maxWidth="sm">
      <DialogTitle>{contact ? "Edit Contact" : "Add Contact"}</DialogTitle>
      <DialogContent>
        <Stack spacing={2} sx={{ mt: 1 }}>
          <TextField name="name" label="Name" value={form.name} onChange={handleChange} fullWidth />
          <TextField name="email" label="Email" value={form.email} onChange={handleChange} fullWidth />
          <TextField name="phone" label="Phone" value={form.phone} onChange={handleChange} fullWidth />
        </Stack>
      </DialogContent>
      <DialogActions>
        <Button onClick={onClose}>Cancel</Button>
        <Button onClick={handleSubmit} variant="contained">
          {contact ? "Update" : "Save"}
        </Button>
      </DialogActions>
    </Dialog>
  );
};

export default ContactForm;
</script>

‚úÖ ContactList.jsx

Reusable searchable list of contacts.


<script type="text/jsx">
import React, { useState } from "react";
import { Box, TextField, Stack, Typography, Button } from "@mui/material";
import AddIcon from "@mui/icons-material/Add";
import ContactCard from "./ContactCard";
import ContactForm from "./ContactForm";

const ContactList = ({ contacts, setContacts }) => {
  const [search, setSearch] = useState("");
  const [openForm, setOpenForm] = useState(false);
  const [selectedContact, setSelectedContact] = useState(null);

  const filtered = contacts.filter(
    (c) =>
      c.name.toLowerCase().includes(search.toLowerCase()) ||
      c.email.toLowerCase().includes(search.toLowerCase())
  );

  const handleSave = (contact) => {
    if (selectedContact) {
      setContacts((prev) =>
        prev.map((c) => (c.email === selectedContact.email ? contact : c))
      );
    } else {
      setContacts((prev) => [...prev, contact]);
    }
    setSelectedContact(null);
  };

  const handleDelete = (contact) => {
    setContacts((prev) => prev.filter((c) => c.email !== contact.email));
  };

  return (
    <Box>
      <Stack direction="row" justifyContent="space-between" alignItems="center" mb={2}>
        <TextField
          label="Search contacts..."
          size="small"
          value={search}
          onChange={(e) => setSearch(e.target.value)}
        />
        <Button
          variant="contained"
          startIcon={<AddIcon />}
          onClick={() => setOpenForm(true)}
        >
          Add Contact
        </Button>
      </Stack>

      {filtered.length === 0 ? (
        <Typography color="text.secondary" align="center">
          No contacts found.
        </Typography>
      ) : (
        <Stack spacing={2}>
          {filtered.map((contact) => (
            <ContactCard
              key={contact.email}
              contact={contact}
              onEdit={(c) => {
                setSelectedContact(c);
                setOpenForm(true);
              }}
              onDelete={handleDelete}
            />
          ))}
        </Stack>
      )}

      <ContactForm
        open={openForm}
        onClose={() => {
          setOpenForm(false);
          setSelectedContact(null);
        }}
        onSave={handleSave}
        contact={selectedContact}
      />
    </Box>
  );
};

export default ContactList;
</script>

‚úÖ ContactManager.jsx

Main page to use all components together.


<script type="text/jsx">
import React, { useState } from "react";
import { Container, Typography, Box } from "@mui/material";
import ContactList from "../components/ContactList";

const ContactManager = () => {
  const [contacts, setContacts] = useState([
    { name: "Paul Dionisio", email: "paul@example.com", phone: "+63 912 345 6789" },
  ]);

  return (
    <Container maxWidth="md" sx={{ mt: 5 }}>
      <Box textAlign="center" mb={4}>
        <Typography variant="h4" fontWeight={600}>
          Contact Management
        </Typography>
        <Typography color="text.secondary">
          Manage your contacts easily
        </Typography>
      </Box>
      <ContactList contacts={contacts} setContacts={setContacts} />
    </Container>
  );
};

export default ContactManager;
</script>
