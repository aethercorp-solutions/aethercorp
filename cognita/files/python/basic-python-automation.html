1. Rename Files in a Folder


import os

folder = "/path/to/folder"
for count, filename in enumerate(os.listdir(folder)):
    old = os.path.join(folder, filename)
    new = os.path.join(folder, f"file_{count}.txt")
    os.rename(old, new)

print("Files renamed successfully!")

2. Send an Email Automatically


import smtplib
from email.mime.text import MIMEText

sender = "you@example.com"
receiver = "friend@example.com"
msg = MIMEText("This is an automated message.")
msg["Subject"] = "Automation Test"
msg["From"] = sender
msg["To"] = receiver

with smtplib.SMTP("smtp.gmail.com", 587) as server:
    server.starttls()
    server.login(sender, "your-app-password")
    server.send_message(msg)

print("Email sent!")

3. Read and Process Excel Data


import pandas as pd

data = pd.read_excel("data.xlsx")
filtered = data[data["Status"] == "Pending"]
filtered.to_excel("pending.xlsx", index=False)
print("Filtered data saved!")

4. Automate a Web Browser (Selenium)


from selenium import webdriver

driver = webdriver.Chrome()
driver.get("https://example.com")
driver.find_element("name", "q").send_keys("Python automation\n")
print(driver.title)
driver.quit()

5. Run a Script Every Hour


import schedule
import time

def job():
    print("Running automation task...")

schedule.every(1).hours.do(job)

while True:
    schedule.run_pending()
    time.sleep(1)
