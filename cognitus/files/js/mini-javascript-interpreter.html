Mini JavaScript interpreter

Below is a beginner-friendly mini JavaScript interpreter you can build step by step. This is not a full JS engine, but it teaches the core ideas used by real interpreters (tokenizer → parser → evaluator).

We’ll support a very small language:


let x = 5
let y = 10
print x + y

1. What our mini interpreter will do

Features

✔ Variables (let x = 5)

✔ Numbers

✔ Addition (+)

✔ Print statement

What it will NOT do (for simplicity)

✘ No loops

✘ No functions

✘ No strings

2. Architecture (like real JS engines)


Code (string)
   ↓
Tokenizer (Lexer)
   ↓
Parser (AST)
   ↓
Interpreter (Evaluator)

3. Step 1 – Tokenizer (Lexer)

The tokenizer breaks text into tokens.

tokenizer.js


function tokenize(input) {
    const tokens = [];
    let current = 0;
    
    while (current < input.length) {
        let char = input[current];
    
        // Skip spaces
        if (char === ' ' || char === '\n') {
        current++;
        continue;
        }
    
        // Numbers
        if (/[0-9]/.test(char)) {
        let value = '';
        while (/[0-9]/.test(char)) {
            value += char;
            char = input[++current];
        }
        tokens.push({ type: 'NUMBER', value });
        continue;
        }
    
        // Identifiers (variables, keywords)
        if (/[a-z]/i.test(char)) {
        let value = '';
        while (/[a-z]/i.test(char)) {
            value += char;
            char = input[++current];
        }
        tokens.push({ type: 'IDENTIFIER', value });
        continue;
        }
    
        // Operators
        if (char === '+' || char === '=') {
        tokens.push({ type: 'OPERATOR', value: char });
        current++;
        continue;
        }
    
        throw new Error('Unknown character: ' + char);
    }
    
    return tokens;
}

4. Step 2 – Parser (AST Builder)

The parser converts tokens into a tree structure.

parser.js


function parse(tokens) {
    let current = 0;
    
    function walk() {
        let token = tokens[current];
    
        // Number
        if (token.type === 'NUMBER') {
        current++;
        return { type: 'NumberLiteral', value: Number(token.value) };
        }
    
        // Variable or keyword
        if (token.type === 'IDENTIFIER') {
        if (token.value === 'let') {
            current++;
            const name = tokens[current++].value; // variable name
            current++; // skip =
            const value = walk();
            return { type: 'VariableDeclaration', name, value };
        }
    
        if (token.value === 'print') {
            current++;
            const expression = walk();
            return { type: 'PrintStatement', expression };
        }
    
        current++;
        return { type: 'Variable', name: token.value };
        }
    
        // Binary expression
        if (token.type === 'OPERATOR' && token.value === '+') {
        current++;
        return {
            type: 'BinaryExpression',
            operator: '+',
            left: walk(),
            right: walk()
        };
        }
    
        throw new Error('Unexpected token: ' + JSON.stringify(token));
    }
    
    const ast = {
        type: 'Program',
        body: []
    };
    
    while (current < tokens.length) {
        ast.body.push(walk());
    }
    
    return ast;
}

5. Step 3 – Interpreter (Evaluator)

This executes the AST.

interpreter.js


function interpret(ast) {
    const env = {}; // variable storage
    
    function evaluate(node) {
        switch (node.type) {
        case 'Program':
            node.body.forEach(evaluate);
            break;
    
        case 'NumberLiteral':
            return node.value;
    
        case 'VariableDeclaration':
            env[node.name] = evaluate(node.value);
            break;
    
        case 'Variable':
            if (!(node.name in env)) {
            throw new Error(`Undefined variable ${node.name}`);
            }
            return env[node.name];
    
        case 'BinaryExpression':
            return evaluate(node.left) + evaluate(node.right);
    
        case 'PrintStatement':
            console.log(evaluate(node.expression));
            break;
    
        default:
            throw new Error('Unknown node type: ' + node.type);
        }
    }
    
    evaluate(ast);
}

6. Step 4 – Wire Everything Together

main.js


const code = `
let x = 5
let y = 10
print x + y
`;

const tokens = tokenize(code);
const ast = parse(tokens);
interpret(ast);

7. Run it on Windows


node main.js

Output:


15

🎉 You just built a mini JavaScript interpreter.
